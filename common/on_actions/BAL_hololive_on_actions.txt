on_actions = {

	on_leave_faction = {
		
		effect = {
			if = {
				limit = { tag = KRO }
				set_variable = { KRO.KRO_council_status = 3 }
				subtract_from_variable = {
					var = number_council_member
					value = 1
				}

			}
		}
