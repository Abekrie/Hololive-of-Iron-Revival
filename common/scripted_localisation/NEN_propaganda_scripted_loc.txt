defined_text = {
	name = NEN_GetDaysRemainingLeft
	text = {
		trigger = {
			OR = {
				check_variable = { NEN_poster_left_icon_frame = 1 }
				check_variable = { NEN_poster_left_icon_frame = 2 }
			}
		}
		localization_key = NEN_poster_no_text
	}
	text = {
		localization_key = NEN_poster_left_days_remaining_text
	}
}
