add_efficiency_idea = {
	if = {
		limit = {
			NOT = {
				has_country_flag = holo_efficiency_idea_in_effect
			}}
		set_country_flag = holo_efficiency_idea_in_effect
		add_ideas = efficiency_i_ns
	}
	
	else_if = {
		limit = {
				has_idea = efficiency_i_ns
			}
		swap_ideas = {
    		remove_idea = efficiency_i_ns
    		add_idea = efficiency_ii_ns
		}
	}
	else_if = {
		limit = {
				has_idea = efficiency_ii_ns
			}
		swap_ideas = {
    		remove_idea = efficiency_ii_ns
    		add_idea = efficiency_iii_ns
		}
	}
	else_if = {
		limit = {
				has_idea = efficiency_iii_ns
			}
		swap_ideas = {
    		remove_idea = efficiency_iii_ns
    		add_idea = efficiency_iv_ns
		}
	}
	else_if = {
		limit = {
				has_idea = efficiency_iv_ns
			}
		swap_ideas = {
    		remove_idea = efficiency_iv_ns
    		add_idea = efficiency_v_ns
		}
	}
}
